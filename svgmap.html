hover回调

// hover回调

    hoverCallback: function(stateData, obj){

click回调

// click回调
$('#ChinaMap6').SVGMap({
	clickColorChange: true,
    clickCallback: function(stateData, obj){
        $('#ClickCallback').html('点击了：'+obj.name);
    },
    unClickCallback: function(stateData, obj){
        $('#UnClickCallback').html('取消点击：'+obj.name);
    }
});
